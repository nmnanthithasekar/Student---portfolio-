document.getElementById("generateBtn").addEventListener("click", () => {
  const name = document.getElementById("userName").value || "Friend";
  const category = document.getElementById("category").value;
  
  const selectedQuotes = quotes[category];
  const randomQuote = selectedQuotes[Math.floor(Math.random() * selectedQuotes.length)];
  
  document.getElementById("quoteDisplay").innerText = randomQuote.replace(/NAME/g, name);
});

document.getElementById("copyBtn").addEventListener("click", () => {
  const quoteText = document.getElementById("quoteDisplay").innerText;
  if (quoteText.trim() === "") {
    alert("No quote to copy!");
    return;
  }
  navigator.clipboard.writeText(quoteText);
  alert("Quote copied to clipboard!");
});
